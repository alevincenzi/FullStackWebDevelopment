./scripts/app.js

    app             /
        header      header.html         HeaderController
        content     home.html           HomeController
        footer      footer.html

    app.aboutus     /aboutus
        content@    aboutus.html        AboutController
    
    app.contactus   /contactus
        content@    contactus.html      ContactController
    
    app.menu        /menu
        content@    menu.html           MenuController
        
    app.dishdetails /menu/:id
        content@    dishdetail.html     DishDetailController
        
    app.favorites   /favorites
        content@    favorites.html      FavoriteController

./scripts/controllers.js

    MenuController
    
        MenuFactory         to query for all dishes
        FavoriteFactory     to save favorite
        
    ContactController
    
        FeedbackFactory     to save contact to restaurant
        
    DishDetailController
    
        MenuFactory         to query for a specific dish
        CommentFactory      to save a comment
        
    HomeController
    
        MenuFactory         to query for featured dish
        CorporateFactory    to query for featured leader
        PromotionFactory    to query for features promotion
        
    AboutController
    
       CorporateFactory     to query for all leaders
       
    FavoriteController
    
       FavoriteFactory      to query for favorite dishes
                            to delete favorite dish
                       
    HeaderController
    
        AuthFactory         to test if user is authenticated
                            to get username
                            to do logout
        - doOpenLogin       login.html      LoginController
        - doLogOut
        
    LoginController
    
        AuthFactory         to do login
        
        - doLogIn
        - doOpenRegister    register.html   RegisterController
        
    RegisterController

        AuthFactory         to add user via registration
        
        - doRegister
        
./scripts/services.js

    MenuFactory             dishes/:dishId
    CommentFactory          dishes/:dishId/comments/:commentId
    PromotionFactory        promotions/:promotionId
    CorporateFactory        leaders/:leaderId
    FavoriteFactory         favorites/:favorite:Id
    FeedbackFactory         feedback/:feedbackId
    AuthFactory             login, logout, register
